"""
Simple test to check Ollama connection.
Run this to diagnose the issue.
"""
import requests

print("=" * 60)
print("OLLAMA CONNECTION TEST")
print("=" * 60)

# Test 1: Is Ollama running?
print("\n1. Testing if Ollama is running...")
try:
    response = requests.get("http://localhost:11434/api/tags", timeout=5)
    print(f"   Status Code: {response.status_code}")
    if response.status_code == 200:
        print("   ‚úÖ Ollama IS running!")
        data = response.json()
        models = data.get("models", [])
        print(f"   üì¶ Found {len(models)} models:")
        for model in models:
            print(f"      - {model.get('name', 'unknown')}")
    else:
        print(f"   ‚ùå Ollama responded with: {response.status_code}")
        print(f"   Response: {response.text}")
except requests.exceptions.ConnectionError:
    print("   ‚ùå Cannot connect to Ollama!")
    print("   üí° Make sure Ollama is running with: ollama serve")
    exit(1)
except Exception as e:
    print(f"   ‚ùå Error: {e}")
    exit(1)

# Test 2: Try /api/generate (old endpoint)
print("\n2. Testing /api/generate endpoint (old format)...")
try:
    response = requests.post(
        "http://localhost:11434/api/generate",
        json={
            "model": "llama3.2:3b",
            "prompt": "Say hello",
            "stream": False
        },
        timeout=30
    )
    print(f"   Status Code: {response.status_code}")
    if response.status_code == 200:
        print("   ‚úÖ /api/generate WORKS!")
        result = response.json()
        print(f"   Response: {result.get('response', '')[:100]}")
    elif response.status_code == 404:
        print("   ‚ö†Ô∏è  /api/generate NOT FOUND (this is your issue)")
    else:
        print(f"   Response: {response.text[:200]}")
except Exception as e:
    print(f"   ‚ùå Error: {e}")

# Test 3: Try /api/chat (new endpoint)
print("\n3. Testing /api/chat endpoint (new format)...")
try:
    response = requests.post(
        "http://localhost:11434/api/chat",
        json={
            "model": "llama3.2:3b",
            "messages": [
                {"role": "user", "content": "Say hello"}
            ],
            "stream": False
        },
        timeout=30
    )
    print(f"   Status Code: {response.status_code}")
    if response.status_code == 200:
        print("   ‚úÖ /api/chat WORKS!")
        result = response.json()
        message = result.get("message", {})
        print(f"   Response: {message.get('content', '')[:100]}")
    elif response.status_code == 404:
        print("   ‚ö†Ô∏è  /api/chat NOT FOUND")
    else:
        print(f"   Response: {response.text[:200]}")
except Exception as e:
    print(f"   ‚ùå Error: {e}")

print("\n" + "=" * 60)
print("TEST COMPLETE")
